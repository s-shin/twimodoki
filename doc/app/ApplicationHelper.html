def hbr(str)
  str = html_escape(str)
  str.gsub(%r\r\n|\r|\n/, "<br />")
end

def latest_tweet(user=nil)
  user = user ? user : @current_user
  return unless user.tweets.length > 0
  yield user.tweets[0]
end
